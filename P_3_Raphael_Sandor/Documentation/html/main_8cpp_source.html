712 void update()
713 {

716  carPosition.x += worldCarSpeed.x;
717  carPosition.z += worldCarSpeed.z;
718 
719  // State machine for the traffic signals using three variables: NS_Signal, WE_Signal, and counter.
720  // Add 1.2 to the counter since the clock seems a little slow.
721  counter += 1.2;
722  if (counter <= fiveSeconds)
723  {
724  NS_Signal = Green;
725  WE_Signal = Red;
726  }
727  else if (counter > fiveSeconds && counter <= sixSeconds)
728  {
729  NS_Signal = Yellow;
730  WE_Signal = Red;
731  }
732  else if (counter > sixSeconds && counter <= twelveSeconds)
733  {
734  NS_Signal = Red;
735  if (counter > elevenSeconds && counter <= twelveSeconds)
736  WE_Signal = Yellow;
737  else
738  WE_Signal = Green;
739  }
740  else if (counter > twelveSeconds)
741  {
742  counter = 0;
743  }
744 }
